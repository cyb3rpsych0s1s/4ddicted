"anchor" variables

one cannot e.g.

  Redscript 
public func GetIDs() -> array<TweakDBID> { ... }

// ❌ WRONG!
public func IsID(id: TweakDBID) -> Bool {
  return ArrayContains(GetIDs(), id);
}
// ✅ do this instead:
public func IsID(id: TweakDBID) -> Bool {
  let ids = GetIDs();
  return ArrayContains(ids, id);
}
